aw-qt -> aw-watcher-afk
aw-notify -> aw-server
aw-watcher-window -> aw-server
aw-watcher-afk -> aw-server
aw-watcher-web -> aw-server

aw-server -> aw-sync
aw-sync -> Sync folder

aw-watcher-afk -> AFK status -> aw-server
aw-watcher-web -> Active tab -> aw-server
